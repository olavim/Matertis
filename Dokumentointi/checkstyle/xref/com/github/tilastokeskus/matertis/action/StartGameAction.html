4   import com.github.tilastokeskus.matertis.core.GameHandler;
5   import com.github.tilastokeskus.matertis.core.Game;
6   import com.github.tilastokeskus.matertis.ui.GameUI;
7   import java.awt.event.ActionEvent;
8   import javax.swing.AbstractAction;
9   import javax.swing.SwingUtilities;
10  
11  /**
12   *
13   * @author tilastokeskus
14   */
15  public class StartGameAction extends AbstractAction {
16  
17      public StartGameAction(String text) {
18          super(text);
19      }
20      
21      @Override
22      public void actionPerformed(ActionEvent e) {
23          Game logic = new Game(10, 20);
24          GameHandler handler = new GameHandler(logic);
25          GameUI ui = new GameUI("New Game", handler);        
26          
27          SwingUtilities.invokeLater(ui);
28          handler.addObserver(ui);
29          handler.startGame();
30      }
31  
32  }
