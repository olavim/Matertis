4   import java.awt.*;
5   import java.awt.event.*;
6   import java.util.ArrayList;
7   import java.util.List;
8   import javax.swing.BorderFactory;
9   import javax.swing.JComponent;
10  import javax.swing.border.Border;
11  import javax.swing.border.EtchedBorder;
12  import javax.swing.event.ChangeEvent;
13  import javax.swing.event.ChangeListener;
14  
15  /**
16   *
17   * @author tilastokeskus
18   */
19  public class KeyBinder extends JComponent {
20      
21      public static final int KEYCODE_EMPTY = -999;
22      
23      private static final Color COLOR_BORDER_HIGHLIGHT = Color.WHITE;
24      private static final Color COLOR_BORDER_SHADOW = new Color(160, 160, 160);
25      
26      private static final Color COLOR_BACKGROUND = new Color(252, 252, 252);
27      private static final Color COLOR_TEXT = new Color(80, 80, 80);
28      private static final Font FONT = new Font(Font.SANS_SERIF, Font.PLAIN, 12);
29      private static final int PADDING = 4;
30      
31      private final int size;
32      private final int id;
33      private final List<ChangeListener> changeListeners;
34      private final ChangeEvent changeEvent;
35      
36      private int keyCode;
37      
38      public KeyBinder(int size, int id, int keyCode) {
39          this.size = size;
40          this.id = id;
41          this.keyCode = keyCode;
42          
43          this.addKeyListener(new KeyBinderKeyListener());
44          this.addMouseListener(new KeyBinderMouseListener());
45          this.addFocusListener(new KeyBinderFocusListener());
46          this.setFont(FONT);        
47          this.setBorder(BorderFactory.createEtchedBorder());
48          
49          this.changeListeners = new ArrayList<>();
50          this.changeEvent = new ChangeEvent(this);
51      }
52      
53      public int getID() {
54          return this.id;
55      }
56      
57      @Override
58      protected void paintComponent(Graphics g) {
59          super.paintComponent(g);
60          Graphics2D g2 = (Graphics2D) g;
61          g2.setRenderingHints(new RenderingHints(
62               RenderingHints.KEY_TEXT_ANTIALIASING,
63               RenderingHints.VALUE_TEXT_ANTIALIAS_ON));
64          
65          g2.setColor(COLOR_BACKGROUND);
66          g2.fillRect(0, 0, this.getWidth(), this.getHeight());
67          
68          g2.setColor(COLOR_TEXT);
69          g2.setFont(this.getFont());
70          
71          FontMetrics metrics = g2.getFontMetrics(this.getFont());
72          
73          int centerY = (this.getHeight() + metrics.getAscent() - PADDING) / 2;
74          String str = KeyEvent.getKeyText(keyCode);
75          g2.drawString(str, PADDING, centerY);
76      }
77      
78      public void setKeyCode(int code) {
79          this.keyCode = code;
80          this.repaint();
81      }
82      
83      public int getKeyCode() {
84          return this.keyCode;
85      }    
86      
87      public void addChangeListener(ChangeListener listener) {
88          this.changeListeners.add(listener);
89      }
90      
91      public void setBorderColor(Color highlight, Color shadow) {
92          this.setBorder(new EtchedBorder(highlight, shadow));
93      }
94      
95      @Override
96      public Dimension getPreferredSize() {
97          int width = this.size * this.getFontMetrics(getFont()).stringWidth("X");
98          int height = this.getFontMetrics(FONT).getHeight();
99          return new Dimension(width + PADDING, height + PADDING);
100     }
101     
102     private void notifyListeners() {
103         for (ChangeListener listener : this.changeListeners) {
104             listener.stateChanged(this.changeEvent);
105         }
106     }
107     
108     private class KeyBinderKeyListener extends KeyAdapter {    
109         @Override
110         public void keyPressed(KeyEvent e) {
111             KeyBinder.this.setKeyCode(e.getKeyCode());
112             KeyBinder.this.notifyListeners();
113         }
114     }
115     
116     private class KeyBinderMouseListener extends MouseAdapter {    
117         @Override
118         public void mousePressed(MouseEvent e) {
119             KeyBinder.this.requestFocusInWindow();
120         }
121     }
122     
123     private class KeyBinderFocusListener implements FocusListener {
124         @Override
125         public void focusGained(FocusEvent e) {
126             Border border = new EtchedBorder(COLOR_BORDER_HIGHLIGHT.darker(),
127                                              COLOR_BORDER_SHADOW.darker());
128             
129             KeyBinder.this.setBorder(border);
130         }
131 
132         @Override
133         public void focusLost(FocusEvent e) {
134             Border border = new EtchedBorder(
135                     COLOR_BORDER_HIGHLIGHT, COLOR_BORDER_SHADOW);
136             KeyBinder.this.setBorder(border);
137         }
138     }
