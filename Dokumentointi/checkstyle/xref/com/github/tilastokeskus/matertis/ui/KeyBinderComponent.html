2   package com.github.tilastokeskus.matertis.ui;
3   
4   import com.github.tilastokeskus.matertis.ui.util.KeyBinder;
5   import java.awt.*;
6   import java.awt.event.*;
7   import java.util.ArrayList;
8   import java.util.List;
9   import javax.swing.BorderFactory;
10  import javax.swing.JComponent;
11  import javax.swing.border.Border;
12  import javax.swing.border.EtchedBorder;
13  import javax.swing.event.ChangeEvent;
14  import javax.swing.event.ChangeListener;
15  
16  /**
17   * Implementation of a "component" KeyBinder. 
18   * 
19   * @author tilastokeskus
20   */
21  public class KeyBinderComponent<T> extends JComponent implements KeyBinder<T> {
22      
23      public static final int KEYCODE_EMPTY = -999;
24      
25      private static final Color COLOR_BORDER_HIGHLIGHT = Color.WHITE;
26      private static final Color COLOR_BORDER_SHADOW = new Color(160, 160, 160);
27      
28      private static final Color COLOR_BACKGROUND = new Color(252, 252, 252);
29      private static final Color COLOR_TEXT = new Color(80, 80, 80);
30      private static final Font FONT = new Font(Font.SANS_SERIF, Font.PLAIN, 12);
31      private static final int PADDING = 4;
32      
33      private final int size;
34      private final T binding;
35      private final List<ChangeListener> changeListeners;
36      private final ChangeEvent changeEvent;
37      
38      private int key;
39      
40      /**
41       * Constructs a KeyBinder component with the given size, binding and initial
42       * key.
43       * 
44       * @param size    size of the component. only affects width.
45       * @param binding the binding to which the key should be bound to.
46       * @param key     the key that should be bound to the object.
47       */
48      public KeyBinderComponent(int size, T binding, int key) {
49          this.size = size;
50          this.binding = binding;
51          this.key = key;
52          this.changeListeners = new ArrayList<>();
53          this.changeEvent = new ChangeEvent(this);
54          
55          this.addKeyListener(new KeyBinderKeyListener());
56          this.addMouseListener(new KeyBinderMouseListener());
57          this.addFocusListener(new KeyBinderFocusListener());
58          this.setFont(FONT);        
59          this.setBorder(BorderFactory.createEtchedBorder());        
60      }
61      
62      @Override
63      protected void paintComponent(Graphics g) {
64          super.paintComponent(g);
65          Graphics2D g2 = (Graphics2D) g;
66          g2.setRenderingHints(new RenderingHints(
67               RenderingHints.KEY_TEXT_ANTIALIASING,
68               RenderingHints.VALUE_TEXT_ANTIALIAS_ON));
69          
70          g2.setColor(COLOR_BACKGROUND);
71          g2.fillRect(0, 0, this.getWidth(), this.getHeight());
72          
73          g2.setColor(COLOR_TEXT);
74          g2.setFont(this.getFont());
75          
76          FontMetrics metrics = g2.getFontMetrics(this.getFont());
77          
78          int centerY = (this.getHeight() + metrics.getAscent() - PADDING) / 2;
79          String str = KeyEvent.getKeyText(key);
80          g2.drawString(str, PADDING, centerY);
81      }
82      
83      @Override
84      public T getBinding() {
85          return this.binding;
86      }
87      
88      /**
89       * {@inheritDoc KeyBinder}
90       * <p>
91       * Notifies listeners of change.
92       * 
93       * @param key {@inheritDoc KeyBinder}
94       */
95      @Override
96      public void setKey(int key) {
97          this.key = key;
98          this.notifyListeners();
99          this.repaint();
100     }
101     
102     @Override
103     public int getKey() {
104         return this.key;
105     }
106     
107     /**
108      * Adds a listener who will be notified whenever the key changes.
109      * 
110      * @param listener 
111      */
112     public void addChangeListener(ChangeListener listener) {
113         this.changeListeners.add(listener);
114     }
115     
116     /**
117      * Changes the etched border's highlight and shadow color.
118      * 
119      * @param highlight the highlight's color.
120      * @param shadow    the shadow's color.
121      */
122     public void setBorderColor(Color highlight, Color shadow) {
123         this.setBorder(new EtchedBorder(highlight, shadow));
124     }
125     
126     private void notifyListeners() {
127         for (ChangeListener listener : this.changeListeners) {
128             listener.stateChanged(this.changeEvent);
129         }
130     }
131     
132     @Override
133     public Dimension getPreferredSize() {
134         int width = this.size * this.getFontMetrics(getFont()).stringWidth("X");
135         int height = this.getFontMetrics(FONT).getHeight();
136         return new Dimension(width + PADDING, height + PADDING);
137     }
138     
139     private class KeyBinderKeyListener extends KeyAdapter {    
140         @Override
141         public void keyPressed(KeyEvent e) {
142             KeyBinderComponent.this.setKey(e.getKeyCode());
143         }
144     }
145     
146     private class KeyBinderMouseListener extends MouseAdapter {    
147         @Override
148         public void mousePressed(MouseEvent e) {
149             KeyBinderComponent.this.requestFocusInWindow();
150         }
151     }
152     
153     private class KeyBinderFocusListener implements FocusListener {
154         @Override
155         public void focusGained(FocusEvent e) {
156             Border border = new EtchedBorder(COLOR_BORDER_HIGHLIGHT.darker(),
157                                              COLOR_BORDER_SHADOW.darker());
158             
159             KeyBinderComponent.this.setBorder(border);
160         }
161 
162         @Override
163         public void focusLost(FocusEvent e) {
164             Border border = new EtchedBorder(
165                     COLOR_BORDER_HIGHLIGHT, COLOR_BORDER_SHADOW);
166             KeyBinderComponent.this.setBorder(border);
167         }
168     }
