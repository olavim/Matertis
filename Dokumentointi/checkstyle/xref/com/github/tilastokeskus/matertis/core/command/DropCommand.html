2   package com.github.tilastokeskus.matertis.core.command;
3   
4   import com.github.tilastokeskus.matertis.core.Game;
5   import com.github.tilastokeskus.matertis.core.GameHandler;
6   import com.github.tilastokeskus.matertis.core.ScoreHandler;
7   
8   /**
9    * Command to drop a tetromino. Command is not executed if game is not running.
10   * 
11   * @author tilastokeskus
12   * @see    Tetromino
13   * @see    Game
14   * @see    ScoreHandler
15   */
16  public class DropCommand implements Command {
17      
18      private final GameHandler gameHandler;
19      
20      /**
21       * Constructs a drop command with the given game and score handler.
22       * @param gameHandler Game handler to be informed.
23       */
24      public DropCommand(GameHandler gameHandler) {
25          this.gameHandler = gameHandler;
26      }
27  
28      /**
29       * Drops the game's falling tetromino, and if rows were cleared, informs the
30       * game handler's score handler of the amount cleared.
31       */
32      @Override
33      public void execute() {
34          if (gameHandler.isRunning()) {
35              gameHandler.getGame().dropFallingTetromino();
36              int cleared = gameHandler.getGame().playRound();
37              gameHandler.getScoreHandler().notifyLinesCleared(cleared);
38          }
39      }
40  
41  }
