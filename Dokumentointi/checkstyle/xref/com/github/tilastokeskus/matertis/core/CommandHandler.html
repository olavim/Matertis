2   package com.github.tilastokeskus.matertis.core;
3   
4   import com.github.tilastokeskus.matertis.core.command.*;
5   import com.github.tilastokeskus.matertis.util.Command;
6   import java.awt.event.KeyEvent;
7   import java.util.HashMap;
8   import java.util.Map;
9   
10  /**
11   * Works as a collection of game commands bound to different identifiers.
12   * Registered commands may be set, unset and executed with the respective id of
13   * the command. The commands are meant to be bound to keyCodes defined in the
14   * {@link KeyEvent} class.
15   * 
16   * @author tilastokeskus
17   */
18  public class CommandHandler {
19      
20      /**
21       * One can use this value to setup a command handler with commands bound to
22       * default identifiers:
23       * <ul>
24       *  <li>KeyEvent.VK_LEFT - move tetromino left.</li>
25       *  <li>KeyEvent.VK_RIGHT - move tetromino right.</li>
26       *  <li>KeyEvent.VK_DOWN - move tetromino down.</li>
27       *  <li>KeyEvent.VK_UP - rotate tetromino.</li>
28       *  <li>KeyEvent.VK_SPACE - drop tetromino.</li>
29       *  <li>KeyEvent.VK_P - pause game.</li>
30       *  <li>KeyEvent.VK_ESCAPE - pause game.</li>
31       * </ul>
32       */
33      public static final int DEFAULT_COMMANDS = 1;
34      
35      private static Map<Integer, Command> getPresetCommands(
36              int presetCommands, GameHandler gameHandler) {
37          
38          switch (presetCommands) {
39              case DEFAULT_COMMANDS:
40                  return getDefaultCommands(gameHandler);
41              default:
42                  throw new IllegalArgumentException();
43          }
44      }
45      
46      private static Map<Integer, Command> getDefaultCommands(
47              GameHandler gameHandler) {
48          Game game = gameHandler.getRegisteredGame();
49          ScoreHandler scoreHandler = gameHandler.getRegisteredScoreHandler();
50          
51          Map<Integer, Command> map = new HashMap<>();
52          map.put(KeyEvent.VK_LEFT,   new LeftCommand(game));
53          map.put(KeyEvent.VK_RIGHT,  new RightCommand(game));
54          map.put(KeyEvent.VK_DOWN,   new DownCommand(game));
55          map.put(KeyEvent.VK_UP,     new RotateCommand(game));
56          map.put(KeyEvent.VK_SPACE,  new DropCommand(game, scoreHandler));
57          map.put(KeyEvent.VK_P,      new PauseCommand(gameHandler));
58          map.put(KeyEvent.VK_ESCAPE, new PauseCommand(gameHandler));
59          
60          return map;
61      }
62  
63      private final Map<Integer, Command> commandBindings;
64      
65      /**
66       * Constructs a command handler with an empty registry. That is, no commands
67       * are bound to anything.
68       */
69      public CommandHandler() {
70          this.commandBindings = new HashMap<>();
71      }
72      
73      /**
74       * Constructs a command handler from an (id, command) map.
75       * @param commands Map containing commands and the identifiers they are
76       *                 bound to.
77       */
78      public CommandHandler(Map<Integer, Command> commands) {
79          this.commandBindings = commands;
80      }
81      
82      /**
83       * Constructs a command handler with the specified preset registry.
84       * @param presetCommands ID of a preset registry.
85       */
86      public CommandHandler(int presetCommands, GameHandler gameHandler) {
87          this.commandBindings = getPresetCommands(presetCommands, gameHandler);
88      }
89      
90      /**
91       * Maps the given integer to the given GameCommand.
92       * @param commandID Identifier of the command.
93       * @param command   Command to be associated with the identifier.
94       */
95      public void registerCommand(int commandID, Command command) {
96          this.commandBindings.put(commandID, command);
97      }
98      
99      /**
100      * Removes any command associated with the given identifier from the
101      * registry.
102      * 
103      * @param commandID The identifier of the command that should be removed.
104      */
105     public void unsetCommand(int commandID) {
106         this.commandBindings.remove(commandID);
107     }
108     
109     /**
110      * Retrieves the command associated with the given identifier.
111      * @param commandID Identifier of the command that should be retrieved.
112      * @return          Command that was associated with the given identifier,
113      *                  or null if there was no command associated with the id.
114      */
115     public Command getCommand(int commandID) {
116         return this.commandBindings.get(commandID);
117     }
118     
119     /**
120      * Returns an (identifier, command) map of the registry.
121      * @return 
122      */
123     public Map<Integer, Command> getCommands() {
124         return this.commandBindings;
125     }
126     
127     /**
128      * Executes a mapped command.
129      * 
130      * @param  commandID   Identifier of the command to be executed.
131      * @return             True if the register contained a command with the
132      *                     given identifier, otherwise false.
133      * @see                Command
134      */
135     public boolean executeCommand(int commandID) {
136         if (this.commandBindings.containsKey(commandID)) {
137             this.commandBindings.get(commandID).execute();
138             return true;
139         } else {
140             return false;
141         }        
142     }
143     
144 }
