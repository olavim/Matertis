2   package com.github.tilastokeskus.matertis.core.command;

4   import com.github.tilastokeskus.matertis.core.Direction;
5   import com.github.tilastokeskus.matertis.core.Game;
6   import com.github.tilastokeskus.matertis.util.Command;
7   
8   /**
9    * Command to move a game's falling tetromino down by one.
10   * 
11   * @author tilastokeskus
12   * @see    Tetromino
13   * @see    Game
14   */
15  public class DownCommand implements Command {
16      
17      private final Game game;
18      
19      /**
20       * Constructs a down command with the given game.
21       * @param game Game to be commanded.
22       */
23      public DownCommand(Game game) {
24          this.game = game;
25      }
26  
27      /**
28       * Moves the game's falling tetromino down by one.
29       */
30      @Override
31      public void execute() {
32          game.moveFallingTetromino(Direction.DOWN);
33      }
34  
35  }
