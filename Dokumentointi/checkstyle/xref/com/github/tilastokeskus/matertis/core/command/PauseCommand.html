2   package com.github.tilastokeskus.matertis.core.command;
3   
4   import com.github.tilastokeskus.matertis.core.GameHandler;
5   
6   /**
7    * Command to toggle the pause state of a game. Command is not executed if game
8    * is over.
9    * 
10   * @author tilastokeskus
11   * @see    Tetromino
12   * @see    GameHandler
13   */
14  public class PauseCommand implements Command {
15  
16      private final GameHandler gameHandler;
17      
18      /**
19       * Constructs a pause command with the given game handler.
20       * @param gameHandler Game handler to be paused and resumed.
21       */
22      public PauseCommand(GameHandler gameHandler) {
23          this.gameHandler = gameHandler;
24      }
25      
26      /**
27       * Toggles the game handler's pause state.
28       */
29      @Override
30      public void execute() {
31          if (!gameHandler.getGame().isGameOver()) {
32              gameHandler.togglePause();
33          }
34      }
35  
36  }
