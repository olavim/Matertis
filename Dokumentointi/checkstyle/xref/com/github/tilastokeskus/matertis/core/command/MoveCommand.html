2   package com.github.tilastokeskus.matertis.core.command;
3   
4   import com.github.tilastokeskus.matertis.core.Direction;
5   import com.github.tilastokeskus.matertis.core.GameHandler;
6   
7   /**
8    * Command to move a game's falling tetromino int some direction by one. Command
9    * is not executed if game is not running.
10   * 
11   * @author tilastokeskus
12   * @see    com.github.tilastokeskus.matertis.core.Tetromino
13   * @see    com.github.tilastokeskus.matertis.core.Direction
14   * @see    com.github.tilastokeskus.matertis.core.GameHandler
15   */
16  public class MoveCommand implements Command {
17      
18      private final GameHandler gameHandler;
19      private final Direction direction;
20      
21      /**
22       * Constructs a move command with the given game handler and direction.
23       * @param gameHandler game handler to be informed.
24       * @param direction   direction to move the tetromino in.
25       */
26      public MoveCommand(GameHandler gameHandler, Direction direction) {
27          this.gameHandler = gameHandler;
28          this.direction = direction;
29      }
30  
31      /**
32       * {@inheritDoc Command}
33       * <p>
34       * Moves the game's falling tetromino in the specified direction by one.
35       */
36      @Override
37      public void execute() {
38          if (gameHandler.isRunning()) {
39              gameHandler.getGame().moveFallingTetromino(direction);
40          }
41      }
42  
43  }
