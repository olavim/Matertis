2   package com.github.tilastokeskus.matertis.core;
3   
4   import java.awt.event.KeyEvent;
5   import java.util.Observable;
6   import java.util.concurrent.Executors;
7   import java.util.concurrent.ScheduledExecutorService;
8   import java.util.concurrent.TimeUnit;
9   import java.util.logging.Level;
10  import java.util.logging.Logger;
11  
12  /**
13   * Provides fundamental functionality to control the flow of a game. This class
14   * controls the game's time and user command related tasks.
15   * <p>
16   * All user commands should be redirected to the game through an instance of
17   * this class.
18   * 
19   * @author tilastokeskus
20   * @see    Game
21   */
22  public class GameHandler extends Observable {
23      
24      private static final Logger LOGGER = Logger.getLogger(Game.class.getName());    
25      private static final long INITIAL_REFRESH_RATE = 1000L;
26  
27      private final Game game;
28      private ScheduledExecutorService roundExecutor;
29      private long refreshRate;
30      
31      /**
32       * Creates a new GameHandler instance and registers the provided game
33       * instance as the one to be handled.
34       * 
35       * @param game A {@link Game} instance to be registered.
36       */
37      public GameHandler(Game game) {
38          this.game = game;
39          this.roundExecutor = Executors.newSingleThreadScheduledExecutor();
40          this.refreshRate = INITIAL_REFRESH_RATE;
41      }
42      
43      public Game getRegisteredGame() {
44          return this.game;
45      }
46      
47      /**
48       * Handles a user command in the form of a keyCode returned by a KeyEvent.
49       * This method interprets the command and redirects a concrete action to the
50       * registered Game instance.
51       * <p>
52       * This method also notifies all observers.
53       * 
54       * @param keyCode An integer key code returned by a KeyEvent.
55       * @see   KeyEvent
56       * @see   java.util.Observer
57       */
58      public void handleKeyCode(int keyCode) {
59          switch (keyCode) {
60              case KeyEvent.VK_LEFT:
61                  game.moveFallingTetromino(Direction.LEFT);
62                  break;
63              case KeyEvent.VK_RIGHT:
64                  game.moveFallingTetromino(Direction.RIGHT);
65                  break;
66              case KeyEvent.VK_DOWN:
67                  game.moveFallingTetromino(Direction.DOWN);
68                  break;
69              case KeyEvent.VK_UP:
70                  game.rotateFallingTetromino();
71                  break;
72              case KeyEvent.VK_SPACE:
73                  game.dropFallingTetromino();
74                  nextRound();
75                  break;
76          }
77  
78          /* notify observers (in practice, tell the game window to refresh) */
79          this.setChanged();
80          this.notifyObservers();
81      }
82      
83      public void startGame() {
84          Runnable roundCmd = new Runnable() {
85              @Override
86              public void run() {
87                  try {
88                      nextRound();
89                      scheduleNextRound(this);
90                  } catch (Throwable e) {
91                      
92                      /* Future silently devours all exceptions, so we log all and
93                       * any exceptions we catch, and then rethrow them.
94                       */
95                      LOGGER.log(Level.SEVERE, null, e);
96                      throw new RuntimeException(e);
97                  }
98              }
99          };
100         
101         this.scheduleNextRound(roundCmd);
102     }
103     
104     private void nextRound() {
105         game.playRound();
106 
107         /* notify observers (in practice, tell the game window to refresh) */
108         this.setChanged();
109         this.notifyObservers();
110     }
111 
112     private void scheduleNextRound(Runnable cmd) {
113         
114         /* schedule a new round if the game is not over */
115         if (!game.gameIsOver()) {
116             roundExecutor.schedule(cmd,
117                                    refreshRate,
118                                    TimeUnit.MILLISECONDS);
119         }
120     }
121     
122 }
