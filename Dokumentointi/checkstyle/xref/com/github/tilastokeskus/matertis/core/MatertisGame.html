2   package com.github.tilastokeskus.matertis.core;
3   
4   import com.github.tilastokeskus.matertis.util.TetrominoFactory;
5   
6   /**
7    *
8    * @author tilastokeskus
9    */
10  public class MatertisGame {
11      
12      private final int[][] grid;
13      
14      private Tetromino fallingTetromino;
15      private Tetromino nextTetromino;
16      private boolean gameIsOver;
17      
18      public MatertisGame(int width, int height) {
19          this.grid = new int[height][width];
20          this.fallingTetromino = TetrominoFactory.getNewTetromino();        
21          this.nextTetromino = TetrominoFactory.getNewTetromino();
22          this.gameIsOver = false;
23      }
24      
25      public void playRound() {
26          if (!this.moveFallingTetromino(Direction.DOWN)) {
27              handleFallenTetromino();
28          }
29      }
30  
31      private void handleFallenTetromino() {
32          GridLogic.setTetromino(grid, fallingTetromino);
33          GridLogic.handleFilledRows(grid);
34          this.setupNewTetromino();
35      }
36      
37      public boolean gameIsOver() {
38          return this.gameIsOver;
39      }
40      
41      public Tetromino getFallingTetromino() {
42          return this.fallingTetromino;
43      }
44      
45      public int[][] getGrid() {
46          return this.grid;
47      }
48      
49      public int getWidth() {
50          return this.grid[0].length;
51      }
52      
53      public int getHeight() {
54          return this.grid.length;
55      }
56      
57      public boolean moveFallingTetromino(Direction direction) {
58          fallingTetromino.move(direction);        
59          if (GridLogic.tetrominoCollides(grid, fallingTetromino)) {
60              fallingTetromino.move(direction.getOpposite());
61              return false;
62          }
63          
64          return true;
65      }
66      
67      public void dropFallingTetromino() {
68          boolean keepDropping;
69          
70          /* move the tetromino down until it hits the ground */
71          do {
72              keepDropping = this.moveFallingTetromino(Direction.DOWN);
73          } while (keepDropping);
74      }
75      
76      public boolean rotateFallingTetromino() {
77          fallingTetromino.rotateCW();
78          
79          /* if the tetromino is rotated into a bad position, rotate it back */
80          if (GridLogic.tetrominoCollides(grid, fallingTetromino)) {
81              fallingTetromino.rotateCCW();
82              return false;
83          }
84          
85          return true;
86      }
87      
88      private void setupNewTetromino() {        
89          if (GridLogic.tetrominoCollides(grid, nextTetromino)) {
90              gameIsOver = true;
91              return;
92          }
93          
94          fallingTetromino = nextTetromino;
95          nextTetromino = TetrominoFactory.getNewTetromino();
96      }
97  
98  }
