2   package com.github.tilastokeskus.matertis.core;
3   
4   /**
5    * Provides fundamental functionality to control the flow of a game. This class
6    * controls the game's time, scoring and user command related tasks.
7    * <p>
8    * All user commands should be redirected to the game through an instance of
9    * this class.
10   * 
11   * @author tilastokeskus
12   * @see    Game
13   * @see    ScoreHandler
14   * @see    CommandHandler
15   */
16  public abstract class AbstractGameHandler {
17      
18      private final Game game;
19      private final ScoreHandler scoreHandler;
20      private CommandHandler commandHandler;    
21      
22      /**
23       * Creates a new GameHandler instance and registers the provided Game as the
24       * one to be handled.
25       * 
26       * @param game           The game to be registered.
27       * @param scoreHandler   The score handler that determines how the scoring
28       *                       of each action should be determined.
29       */
30      public AbstractGameHandler(Game game, ScoreHandler scoreHandler) {
31          this.game = game;
32          this.scoreHandler = scoreHandler;
33      }
34      
35      public Game getRegisteredGame() {
36          return this.game;
37      }
38      
39      public ScoreHandler getRegisteredScoreHandler() {
40          return this.scoreHandler;
41      }
42      
43      public CommandHandler getRegisteredCommandHandler() {
44          return this.commandHandler;
45      }
46      
47      public void registerCommandHandler(CommandHandler commandHandler) {
48          this.commandHandler = commandHandler;
49      }
50      
51      /**
52       * Starts the game.
53       */
54      public abstract void startGame();
55      
56      /**
57       * Ends the running game.
58       */
59      public abstract void terminateGame();
60      
61      /**
62       * Toggles the pause-state of the game. When paused, the game should not
63       * advance, and should not accept user commands.
64       */
65      public abstract void togglePause();
66      
67      /**
68       * Returns whether or not the game is currently paused.
69       * 
70       * @return True if the game is paused, false if not.
71       */
72      public abstract boolean isPaused();
73      
74      /**
75       * Executes a user command that is bound to the provided keyCode returned by
76       * a KeyEvent. This method interprets the command and redirects a concrete
77       * action to the registered Game instance.
78       * <p>
79       * This method also notifies all observers.
80       * 
81       * @param keyCode An integer key code returned by a KeyEvent. Available
82       * keyCodes and commands mapped to them:
83       * <ul>
84       *  <li>KeyEvent.VK_LEFT - moves the falling tetromino left.</li>
85       *  <li>KeyEvent.VK_RIGHT - moves the falling tetromino right.</li>
86       *  <li>KeyEvent.VK_DOWN - moves the falling tetromino down.</li>
87       *  <li>KeyEvent.VK_UP - rotates the falling tetromino.</li>
88       *  <li>KeyEvent.VK_SPACE - drops the falling tetromino.</li>
89       *  <li>KeyEvent.VK_P - pauses the game.</li>
90       *  <li>KeyEvent.VK_ESCAPE - pauses the game.</li>
91       * </ul>
92       * 
93       * @return True if the command was executed successfully, otherwise false.
94       * 
95       * @see KeyEvent
96       * @see java.util.Observer
97       */
98      public abstract boolean executeCommand(int keyCode);
99  
100 }
