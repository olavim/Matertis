4   /**
5    * Provides functionality to modify 2-dimensional integer matrices to include
6    * data of tetrominoes' whereabouts in relation to the grid. This functionality
7    * includes "inserting" a tetromino to the grid and dropping all tetrominoes
8    * above a filled row down.
9    * 
10   * @author tilastokeskus
11   * @see    Tetromino
12   */
13  public final class GridLogic {
14      private GridLogic() { }
15      
16      private static final int EMPTY = 0;
17      
18      /**
19       * Tells whether  a tetromino is in collision with a non-empty cell in the
20       * provided matrix. An empty cell should be represented by zero (0).
21       * <p>
22       * The matrix should be row-indiced. That is, grid[0] should return the
23       * first row in the matrix, and grid[3][6] should return the 7th cell of the
24       * 4th row, from top left to bottom right.
25       * 
26       * @param grid      A 2-dimensional integer matrix holding information of
27       *                  occupied and unoccupied cells.
28       * @param tetromino A Tetromino object whose collision should be tested
29       *                  against the grid.
30       * @return          True if the tetromino collides with the grid, otherwise
31       *                  false.
32       * @see             Tetromino
33       */
34      public static boolean tetrominoCollides(int[][] grid, Tetromino tetromino) {
35          
36          /* iterate through the tetromino's layout */
37          for (int row = 0; row < tetromino.getLayout().length; row++) {
38              for (int col = 0; col < tetromino.getLayout()[0].length; col++) {
39                  if (tetromino.getLayout()[row][col] == EMPTY) {
40                      continue;
41                  }
42                  
43                  int x = col + tetromino.getX();
44                  int y = row + tetromino.getY();
45                  
46                  /* if the piece's location in the grid is not out of bounds, or
47                   * if in that location is a piece of some other tetromino,
48                   * return true.
49                   */
50                  if (isOutOfBounds(grid, x, y) || grid[y][x] != EMPTY) {
51                      return true;
52                  }
53              }
54          }
55          
56          return false;
57      }
58      
59      /**
60       * Tells whether or not the provided x and y are both inside a grid's
61       * bounds.
62       * 
63       * @param grid A 2-dimensional integer matrix holding information of
64       *             occupied and unoccupied cells.
65       * @param x    The x of the point.
66       * @param y    The y of the point.
67       * @return     True if the point is out of bounds, otherwise false.
68       */
69      public static boolean isOutOfBounds(int[][] grid, int x, int y) {
70          return x < 0 || x >= grid[0].length || y < 0 || y >= grid.length;
71      }
72      
73      /**
74       * Adds a tetromino's layout in the grid according to the tetromino's
75       * position.
76       * <p>
77       * This method makes no checks of whether or not the tetromino
78       * collides with the grid, or if part of it is outside of the grid's bounds.
79       * If only a part of the tetromino is in the grid's bounds, the part that is
80       * in the grid's bounds will be added to the grid nonetheless.
81       * 
82       * @param grid      A 2-dimensional integer matrix holding information of
83       *                  occupied and unoccupied cells.
84       * @param tetromino A Tetromino object that should be added to the grid.
85       * @see             Tetromino
86       */
87      public static void setTetromino(int[][] grid, Tetromino tetromino) {
88          int id = tetromino.getIdentifier();
89          
90          /* iterate through the tetromino's layout */
91          for (int row = 0; row < tetromino.getLayout().length; row++) {
92              for (int col = 0; col < tetromino.getLayout()[0].length; col++) {
93                  if (tetromino.getLayout()[row][col] == EMPTY) {
94                      continue;
95                  }
96                  
97                  int x = col + tetromino.getX();
98                  int y = row + tetromino.getY();
99                  
100                 if (!isOutOfBounds(grid, x, y)) {
101                     grid[y][x] = tetromino.getLayout()[row][col] * id;
102                 }
103             }
104         }
105     }
106     
107     /**
108      * Searches for filled rows from the grid, top to bottom, and on each filled
109      * row it drops all rows on top of it by one.
110      * @param grid A 2-dimensional integer matrix holding information of
111      *             occupied and unoccupied cells.
112      * @see        #dropRowsDownFromIndex(int[][], int)
113      *             dropRowsDownFromIndex(int[][], int)
114      * @see        #rowIsFilled(int[])
115      *             rowIsFilled(int[])
116      */
117     public static void handleFilledRows(int[][] grid) {
118         for (int rowIndex = 0; rowIndex < grid.length; rowIndex++) {
119             if (rowIsFilled(grid[rowIndex])) {
120                 dropRowsDownFromIndex(grid, rowIndex);
121             }
122         }
123     }
124     
125     /**
126      * Removes the given row from the grid and adds an empty row on top of the
127      * grid.
128      * 
129      * @param grid     A 2-dimensional integer matrix holding information of
130      *                 occupied and unoccupied cells.
131      * @param rowIndex The index of the row in the grid.
132      */
133     public static void dropRowsDownFromIndex(int[][] grid, int rowIndex) {        
134         for (int col = 0; col < grid[0].length; col++) {
135             
136             /* iterate from rowIndex to the top of the grid */
137             for (int row = rowIndex; row > 0; row--) {
138                 
139                 /* take the value of the above cell, and set it to this cell */
140                 grid[row][col] = grid[row - 1][col];
141             }
142             
143             /* make this column's top row empty */
144             grid[0][col] = EMPTY;
145         }
146     }
147 
148     /**
149      * Tells whether the given row is fully occupied. That is, in the row there
150      * is not a single zero (0) element.
151      * 
152      * @param row An integer array representing the row that should be examined.
153      * @return    True if the row is fully occupied, otherwise false.
154      */
155     public static boolean rowIsFilled(int[] row) {
156         for (int n : row) {
157             
158             /* if some element in the row is empty, return false */
159             if (n == EMPTY) {
160                 return false;
161             }
162         }
163         
164         return true;
165     }
