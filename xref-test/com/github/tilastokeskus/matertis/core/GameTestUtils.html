12  public class GameTestUtils {
13      
14      private static final Logger LOGGER =
15              Logger.getLogger(GameTestUtils.class.getName());
16      
17      public static void setGameFallingTetromino(Game game, Tetromino tetromino) {
18          try {
19              Field f = game.getClass().getDeclaredField("fallingTetromino");
20              f.setAccessible(true);
21              f.set(game, tetromino);
22          } catch (IllegalArgumentException | IllegalAccessException
23                  | NoSuchFieldException | SecurityException ex) {
24              LOGGER.log(Level.SEVERE, null, ex);
25          }
26      }
