4   import java.awt.Container;
5   import java.awt.Window;
6   import java.awt.event.ActionEvent;
7   import java.awt.event.ActionListener;
8   import javax.swing.JButton;
9   import javax.swing.JDialog;
10  import javax.swing.JLabel;
11  import javax.swing.JPanel;
12  import javax.swing.SwingUtilities;
13  import javax.swing.UIManager;
14  import net.miginfocom.swing.MigLayout;
15  
16  /**
17   *
18   * @author tilastokeskus
19   */
20  public class ErrorDialog extends JDialog {
21      
22      public static void showMsg(String message) {
23          showMsg(null, message);
24      }
25      
26      public static void showMsg(final Window parent, final String message) {
27          SwingUtilities.invokeLater(new Runnable() {
28              @Override
29              public void run() {
30                  ErrorDialog dialog = new ErrorDialog(parent, message);
31                  dialog.showDialog();
32              }
33          });
34      }
35      
36      private final Window parent;
37      private final String message;
38      
39      private ErrorDialog(Window parent, String message) {
40          super(parent, "Error", JDialog.DEFAULT_MODALITY_TYPE);
41          this.parent = parent;
42          this.message = message;
43      }
44      
45      public void showDialog() {
46          this.setResizable(false);
47          
48          this.addContents(this.getContentPane());
49          
50          this.pack();
51          this.setLocationRelativeTo(this.parent);
52          this.setVisible(true);
53      }
54      
55      private void addContents(Container container) {
56          container.setLayout(new MigLayout());
57          
58          JLabel icon = new JLabel(UIManager.getIcon("OptionPane.errorIcon"));
59          JButton okButton = new JButton("OK");
60          okButton.addActionListener(new ActionListener() {
61              @Override
62              public void actionPerformed(ActionEvent e) {
63                  dispose();
64              }
65          });
66          
67          container.add(icon, "gap top 5");
68          container.add(new JLabel(message), "wrap, gap left 5, gap right 10");
69          container.add(okButton, "span, center, gap 10");        
70      }
71  
72  }
