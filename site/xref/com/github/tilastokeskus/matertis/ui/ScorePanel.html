3   
4   import com.github.tilastokeskus.matertis.core.ScoreHandler;
5   import java.awt.Color;
6   import java.awt.Dimension;
7   import java.awt.Font;
8   import javax.swing.JLabel;
9   import javax.swing.JPanel;
10  import net.miginfocom.swing.MigLayout;
11  
12  /**
13   * Panel showing the current score and level of a game.
14   * 
15   * @author tilastokeskus
16   */
17  public class ScorePanel extends JPanel {
18      
19      private final static Font FONT_SCORE =
20              new Font(Font.MONOSPACED, Font.BOLD, 12);
21      
22      private final static Font FONT_LABEL =
23              new Font(Font.SANS_SERIF, Font.BOLD, 12);
24      
25      private final JLabel scoreLabel;
26      private final JLabel levelLabel;
27      
28      public ScorePanel() {
29          this.setLayout(new MigLayout("insets 10 0, wrap 1", "[grow]", "[]"));
30          
31          this.scoreLabel = new JLabel("0");
32          this.scoreLabel.setFont(FONT_SCORE);
33          this.scoreLabel.setForeground(new Color(180, 180, 180));
34          
35          this.levelLabel = new JLabel("0");
36          this.levelLabel.setFont(FONT_SCORE);
37          this.levelLabel.setForeground(new Color(180, 180, 180));
38          
39          this.addContents();
40      }
41      
42      private void addContents() {
43          MigLayout scoreLayout = new MigLayout("insets 0", "[grow]", "[grow]");
44          JPanel scorePanel = new BorderedPanel(
45                  new RoundedLineBorder(15, Color.BLACK));
46          scorePanel.setLayout(scoreLayout);
47          scorePanel.setBackground(new Color(40, 40, 40));
48          scorePanel.setPreferredSize(new Dimension(80, 30));
49          scorePanel.add(scoreLabel, "center");
50          
51          MigLayout levelLayout = new MigLayout("insets 0", "[grow]", "[grow]");
52          JPanel levelPanel = new BorderedPanel(
53                  new RoundedLineBorder(15, Color.BLACK));
54          levelPanel.setLayout(levelLayout);
55          levelPanel.setBackground(new Color(40, 40, 40));
56          levelPanel.setPreferredSize(new Dimension(80, 30));
57          levelPanel.add(levelLabel, "center");
58          
59          JLabel score = new JLabel("Score");
60          score.setFont(FONT_LABEL);
61          score.setForeground(new Color(200, 200, 200));
62          JLabel level = new JLabel("Level");
63          level.setFont(FONT_LABEL);
64          level.setForeground(new Color(200, 200, 200));
65          
66          this.add(score);
67          this.add(scorePanel, "grow");
68          this.add(level, "gapy 10");
69          this.add(levelPanel, "grow");
70      }
71      
72      /**
73       * Updates the score according to the given scoreHandler.
74       * 
75       * @param scoreHandler Score handler according to which the score and level
76       *                     should be drawn.
77       */
78      public void setScore(ScoreHandler scoreHandler) {
79          this.scoreLabel.setText("" + scoreHandler.getScore());
80          this.levelLabel.setText("" + scoreHandler.getLevel());
81      }
82  
83  }
